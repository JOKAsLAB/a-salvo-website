    <script>
        document.addEventListener("DOMContentLoaded", function () {
            function toggleInput(radioName, inputName) {
                const radios = document.querySelectorAll(input[name="${radioName}"]);
                const input = document.querySelector(input[name="${inputName}"]);

                radios.forEach(radio => {
                    radio.addEventListener("change", function () {
                        input.disabled = this.value !== "outro";
                        if (input.disabled) input.value = ""; // Limpa o campo se for desativado
                    });
                });

                // Inicializa desativado
                input.disabled = true;
            }

            toggleInput("familia_pessoal", "familia_pessoal_outro");
            toggleInput("familia_telefone", "familia_telefone_outro");
        });

        document.getElementById("addTestemunha").addEventListener("click", function () {
            let container = document.getElementById("testemunhasContainer");
            let index = container.getElementsByClassName("testemunha").length + 1;
            let div = document.createElement("div");
            div.classList.add("testemunha");
            div.innerHTML = `
                <p style="text-align: center; font-weight: bold">Testemunha ${index}</p>
                <label>Nome: <input type="text" name="nome_testemunha${index}"></label>
                <label>Morada: <input type="text" name="morada_testemunha${index}"></label>
                <label>Código Postal: <input type="text" name="codigo_postal_testemunha${index}"></label>
                <label>Idade: <input type="number" name="idade_testemunha${index}"></label>
                <label>Nº Telefone: <input type="text" name="telefone_testemunha${index}"></label>
                <label>
                    Sexo:
                    <select name="sexo_testemunha${index}">
                        <option value="M">Masculino</option>
                        <option value="F">Feminino</option>
                    </select>
                </label>
                <label>Nacionalidade: <input type="text" name="nacionalidade_testemunha${index}"></label>
                <label>Assinatura: <input type="text" name="assinatura_testemunha${index}"></label>
                <button type="button" class="remove-testemunha">Remover Testemunha</button>
            `;
            container.appendChild(div);

            div.querySelector(".remove-testemunha").addEventListener("click", function () {
                div.remove();
            });
        });
    </script>
